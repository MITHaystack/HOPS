185  static std::vector< std::string > split_version_string(std::string vstr, char delim)
186  {
187  std::vector<std::string> parts;
188  std::size_t start = 0;
189  std::size_t end = 0;
190  while((end = vstr.find(delim, start)) != std::string::npos)
191  {
192  parts.push_back(vstr.substr(start, end - start));
193  start = end + 1;
194  }
195  parts.push_back(vstr.substr(start));
196  return parts;
197  }
